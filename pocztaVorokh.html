Funkcja: oblicza średnią i zwraca liczbę najbliższą średniej double najblizsza(double a, double b, double c, double d, double e) { double liczby[5] = {a, b, c, d, e}; double srednia = (a + b + c + d + e) / 5.0; double naj = liczby[0]; double min_roznica = fabs(liczby[0] - srednia); for (int i = 1; i < 5; i++) { double roznica = fabs(liczby[i] - srednia); if (roznica < min_roznica) { min_roznica = roznica; naj = liczby[i]; } } cout << "Średnia: " << srednia << endl; return naj; } // 3. Funkcja: odwraca ciąg znaków string odwroc(string tekst) { string wynik = ""; for (int i = tekst.length() - 1; i >= 0; i--) { wynik = wynik + tekst[i]; } return wynik; } // --- Funkcja główna --- int main() { string imie; cout << "Podaj imie: "; cin >> imie; cout << plec(imie) << endl << endl; double a, b, c, d, e; cout << "Podaj 5 liczb rzeczywistych: "; cin >> a >> b >> c >> d >> e; cout << "Liczba najblizsza sredniej to: " << najblizsza(a, b, c, d, e) << endl << endl; cin.ignore(numeric_limits::max(), '\n'); // czyszczenie bufora string tekst; cout << "Podaj ciag znakow: "; getline(cin, tekst); cout << "Odwrocony ciag: " << odwroc(tekst) << endl; return 0; }
